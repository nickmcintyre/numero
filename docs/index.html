A friendly and intuitive math library for p5.js

This addon library for p5.js turns the "software sketchbook" into a beginner-friendly environment for technical computing. It provides the following features:

  • A tensor object (similar to NumPy arrays)
  • A computer algebra system (similar to SymPy)
  • A grammar of graphics (similar to ggplot2)
  • A supercharged p5.Table (similar to pandas DataFrames)
  • A machine learning API (similar to scikit-learn -- coming soon!)
  • A drawing turtle (because they're awesome)

The library is written in TypeScript and uses Day.js, Math.js, and TensorFlow.js under the hood. It bundles p5.dataframe, TurtleGFX, and wildflower.

Usage

Plotting

View the plotting example.

let iris
let plot

function preload() {
  iris = loadTable('iris.csv', 'csv', 'header')
}

function setup() {
  createCanvas(400, 400)

  plot = createPlot(iris)
}

function draw() {
  plot.title('Iris sepals')
  plot.xlabel('Width')
  plot.ylabel('Height')
  plot.point({
    x: 'SepalWidth',
    y: 'SepalLength',
  })
}

Tables

View the tables example.

let iris

function preload() {
  iris = loadTable('iris.csv', 'csv', 'header')
}

function setup() {
  noCanvas()
  iris.inferTypes()
  // print('The iris dataset')
  // iris.print()
  print('Iris dataset summary by column')
  let summary = iris.describe()
  summary.print()
  print('Mean by species')
  let mean = iris.groupby('Species').mean()
  mean.print()
}
